#include <iostream>

string Tens(int num);
string Ones(int num);

int main()
{
 int num;
 string ones, tens;

while (true){
 
 cout << "Enter number: ";
 cin >> num;

 ones = Ones(num);

 }  

return 0;}

string tens(int num)
{
 string tens;
 num = num / 10;
 
 if (num == 0) {
 	tens = "";
	}
 if (num == 2) {
 	tens = "Twenty ";
	}
 if (num == 3) {
 	tens = "Thirty ";
	}
 if (num == 4) {
 	tens = "Forty ";
	}
 if (num == 5) {
 	tens = "Fifty ";
	}
 if (num == 6) {
 	tens = "Sixty ";
	}
 if (num == 7) {
 	tens = "Seventy ";
	}
 if (num == 8) {
 	tens = "Eighty ";
	}
 if (num == 9) {
 	tens = "Ninty";
	}
 return tens;
}
